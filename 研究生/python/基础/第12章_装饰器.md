# 装饰器

简单的原理，因为函数可以作为参数所以为了封闭开放原则我们可以使用装饰器来给函数添加功能

```python
def test_time(x):
    def inner():
        start_time = time.time()
        x()
        end_time = time.time()
        printf("xxx")
    return inner

@test_time #index = test_time(index)
def index():
    time.sleep(2)
    print(xxxx)
index()
```

### 带返回值的装饰器

```python
def test_time(x):
    def inner():
        start_time = time.time()
        x()
        end_time = time.time()
        printf("xxx")
    return inner

@test_time #index = test_time(index)
def index():
    time.sleep(2)
    print(xxxx)
    return true
a=index() #a=none
```

因为实际执行的是innner所以没有返回true

```python
def test_time(x):
    def inner():
        start_time = time.time()
        ret = x()
        end_time = time.time()
        printf("xxx")
        return ret
    return inner

@test_time #index = test_time(index)
def index():
    time.sleep(2)
    print(xxxx)
    return true
a=index() #a=true
```

所以要再inner中返回index的返回值

### 被装饰函数加参数

```python
def test_time(x):
    def inner():
        start_time = time.time()
        ret = x()
        end_time = time.time()
        printf("xxx")
        return ret
    return inner

@test_time #index = test_time(index)
def index(n):
    time.sleep(2)
    print(xxxx{n})
    return true
a=index('123') #error
```

因为实际是执行的是inner inner没有参数所有出现错误

```python
def test_time(x):
    def inner(*args,**warg):
        start_time = time.time()
        ret = x(*args,**warg)
        end_time = time.time()
        printf("xxx")
        return ret
    return inner

@test_time #index = test_time(index)
def index(n):
    time.sleep(2)
    print(xxxx{n})
    return true
a=index('123') #error
```

这样无论几个参数都可以正常执行

### 带参数的装饰器

 ```python
 def warp_out(index):
     print(index)
     def out(n):
         print(n)
         def wrapper():
             n()
             print(n)
         return wrapper
     return out
 
 
 
 @warp_out('腾讯')
 def qq():
     print("qq")
 
 
 qq()
 ```

每次添加参数需要嵌套一层
